a=int(input("Enter 1st number:"))
b=int(input("Enter 2nd number:"))
c=input("Enter the Operation +,-,/,*:")
print("The result is:",end='')
if c=='+':
    print(a+b)
elif c=='-':
    print(a-b)
elif c=='/':
    print(a*b)
else:
    print("Error:Wrong operator entered")
